Question 1

Add Song using useMutation

import { gql, useMutation } from "@apollo/client";

const ADD_SONG = gql`
mutation AddSong($title:String!, $artist:String!){
  addSong(title:$title, artist:$artist){
    id
    title
    artist
  }
}
`;

const [addSong] = useMutation(ADD_SONG);

const handleSubmit = () => {

addSong({

variables:{
title:"Believer",
artist:"Imagine Dragons"
}

});

};

Question 2

Update Song Title

const UPDATE_SONG = gql`
mutation UpdateSong($id:ID!, $title:String!){
updateSong(id:$id,title:$title){
id
title
}
}
`;

const [updateSong] = useMutation(UPDATE_SONG);

updateSong({

variables:{
id:1,
title:"New Song Title"
},

refetchQueries:["GetSongs"]

});

Question 4

Optimistic UI Update

const [addSong] = useMutation(ADD_SONG);

addSong({

variables:{
title:"Perfect",
artist:"Ed Sheeran"
},

optimisticResponse:{

addSong:{

__typename:"Song",

id:"temp-id",

title:"Perfect",

artist:"Ed Sheeran"

}

}

});

The song appears immediately in the UI before the server responds, making the application feel faster.

Question 5

Automatically Update Song Count

const GET_SONGS = gql`
query{
songs{
id
title
}
}
`;

const { data } = useQuery(GET_SONGS);

return(







Total Songs :
{data?.songs.length}







);

Explanation

  • Mutations update the backend data.
  • Apollo Client updates its cache automatically or by refetching queries.
  • The React component re-renders automatically.
  • The total song count updates immediately after adding or deleting a song.
