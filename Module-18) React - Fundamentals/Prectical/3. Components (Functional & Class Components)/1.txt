import React, { Component } from "react";

// Functional Component
function Greeting(props) {
  return <h1>Hello, {props.name}!</h1>;
}

// Class Component
class WelcomeMessage extends Component {
  render() {
    return <h1>Welcome to React!</h1>;
  }
}

function App() {
  return (
    <div>
      <Greeting name="Rahul" />
      <WelcomeMessage />
    </div>
  );
}

export default App;
