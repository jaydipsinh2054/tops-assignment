1. Find Orders Placed in the Last 30 Days ($match)

db.orders.aggregate([
    {
        $match: {
            orderDate: {
                $gte: new Date(
                    Date.now() - (30 * 24 * 60 * 60 * 1000)
                )
            }
        }
    }
]);
